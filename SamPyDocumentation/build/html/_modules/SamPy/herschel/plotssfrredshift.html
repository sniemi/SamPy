    xe = data[:,0][mask]
    ye = data[:,1][mask]
    # disks
    disks = (data[:,2] <= bulge) & (data[:,3] > 0.5)
    xd = data[:,0][disks]
    yd = data[:,1][disks]
    #IR bright

    xir = data[:,0][mask]
    yir = data[:,1][mask]
